program DemoRepeat;

const
  LIMITE = 5;
  TEXTO = 'IteraciÃ³n:';

var
  numero: INTEGER;
  total: INTEGER;

function Cuadrado(n: INTEGER): INTEGER;
begin
  Cuadrado := n * n;
end;

begin
  numero := 1;
  total := 0;

  repeat begin
    writeln(TEXTO, ' ', numero);
    total := total + Cuadrado(numero);
    numero := numero + 1;
  end until numero > LIMITE;

  writeln('Suma de cuadrados: ', total);
end.
